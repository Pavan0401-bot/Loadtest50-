pipeline {
    agent any

    environment {
        IMAGE_NAME = 'pavan9908/jmetertester1:v1.0'
        TEST_REPO = 'https://github.com/Pavan0401-bot/AB_AS_Sampledata50users.git'
    }

    stages {

        stage('Checkout Tests') {
            steps {
                git branch: 'main',
                    credentialsId: 'GitHub-PAT',
                    url: TEST_REPO
            }
        }

        stage('Run JMeter Tests') {
            steps {
                bat """
                docker run --rm -v %cd%\\:/tests -w /tests ${IMAGE_NAME} ^
                -n -t Test Plan.jmx -l results.jtl
                """
            }
        }

        stage('Archive Results') {
            steps {
                archiveArtifacts artifacts: 'results.jtl, report/**', allowEmptyArchive: true
            }
        }

        stage('Push Results to GitHub') {
            steps {
                withCredentials([
                    usernamePassword(
                        credentialsId: 'GitHub-PAT',
                        usernameVariable: 'GIT_USER',
                        passwordVariable: 'GIT_PASS'
                    )
                ]) {
                    bat """
                    git config --global user.email "pavane2424@gmail.com"
                    git config --global user.name "Pavan0401-bot"

                    rem Pull latest changes first
                    git pull https://%GIT_USER%:%GIT_PASS%@github.com/Pavan0401-bot/AB_AS_Sampledata50users.git main

                    rem Create results folder if not exists
                    if not exist results mkdir results

                    rem Append build number to results file and folder
                    set BUILD_NUM=%BUILD_NUMBER%
                    copy /Y \\results.jtl results\\results_%BUILD_NUM%.jtl
                    xcopy /E /I /Y report results\\report_%BUILD_NUM%

                    git add results\\

                    git commit -m "Automated: Add latest JMeter results for build #%BUILD_NUM%" || echo Nothing to commit

                    git push https://%GIT_USER%:%GIT_PASS%@github.com/Pavan0401-bot/AB_AS_Sampledata50users.git main
                    """
                }
            }
        }
    }
}
